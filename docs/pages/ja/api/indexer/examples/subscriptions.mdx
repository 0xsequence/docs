## MOCKDATA + ja-translated Subscribe to Blockchain Events
ja-translated You can subscribe to different blockchain events in real time using
subscription endpoints. Use filters to listen for particular contract
addresses, account addresses, and/or token IDs.

### MOCKDATA + ja-translated Subscribing to Events
ja-translated _Sequence Indexer `SubscribeEvents` Method:_
- MOCKDATA + ja-translated Request: `POST /rpc/Indexer/SubscribeEvents`
- ja-translated Content-Type: `application/json`
- MOCKDATA + ja-translated Body (in JSON):
  - MOCKDATA + ja-translated `Filters` (\[]object) -- an array of filters
    - ja-translated `contractAddresses` (\[]string) -- a ERC20 / ERC721 / ERC1155 contract address
    - MOCKDATA + ja-translated `accounts` (\[]string) -- wallet addresses
    - ja-translated `tokenIDs` (\[]int) _optional_ -- an array of token ids
    - MOCKDATA + ja-translated `events` (\[]string) -- an array of event names
    - ja-translated `topicHashes` (\[]string) -- an array of topic hashes

<br />

MOCKDATA + ja-translated One of `contractAddresses`, `accounts` must be provided in the filter.

```go [Go]
import (
  "context"
  "log"

  "github.com/0xsequence/go-sequence/indexer"
  "github.com/0xsequence/go-sequence/lib/prototyp"
)

func SubscribeEvents() {
  seqIndexer := indexer.NewIndexer("https://polygon-indexer.sequence.app", "c3bgcU3LkFR9Bp9jFssLenPAAAAAAAAAA")

  reader, err := seqIndexer.SubscribeEvents(
    context.Background(),
    &indexer.EventFilter{
      ContractAddresses: []prototyp.Hash{prototyp.HashFromString("0x3c499c542cEF5E3811e1192ce70d8cC03d5c3359")},
    },
  )
  if err != nil {
    log.Fatalf("SubscribeEvents: %v", err)
  }

  for {
    event, err := reader.Read()
    if err != nil {
      log.Fatalf("Read: %v", err)
    }

    log.Println("Event", event)
  }
}
```

```typescript [TypeScript]
// Works in both a Webapp (browser) or Node.js:
import { SequenceIndexer, WebrpcError } from '@0xsequence/indexer'

const indexer = new SequenceIndexer('https://polygon-indexer.sequence.app', 'c3bgcU3LkFR9Bp9jFssLenPAAAAAAAAAA')

const req = {
    filter: {
      contractAddresses: ['0x3c499c542cEF5E3811e1192ce70d8cC03d5c3359'],
    },
}

const options = {
  onMessage: (msg: any) => {
    console.log('msg', msg)
  },
  onError: (err: WebrpcError) => {
    console.error('err', err)
  }
}

await indexer.subscribeEvents(req, options)
```
