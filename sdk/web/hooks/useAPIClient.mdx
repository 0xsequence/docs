## Import

```tsx
import { useAPIClient } from '@0xsequence/hooks'
```

## Usage

```tsx
import { useAPIClient } from '@0xsequence/hooks'

function ExchangeRateWidget() {
  const apiClient = useAPIClient()
  
  // Example: Get exchange rate
  const fetchExchangeRate = async () => {
    const rate = await apiClient.getExchangeRate({
      toCurrency: 'EUR'
    })
  }
  
  // Example: Get swap quote
  const fetchSwapQuote = async () => {
    const quote = await apiClient.getSwapQuote({
      chainId: 1,
      userAddress: '0x1234...',
      sellCurrencyAddress: '0x...',
      buyCurrencyAddress: '0x...',  
      sellAmount: '1' 
    })
  }
}
```

## Return Type

```tsx
SequenceAPIClient
```

The hook returns a memoized instance of the Sequence API client, which provides access to various API endpoints and services.

### Common API Methods

#### getExchangeRate

```tsx
getExchangeRate(params: {
  toCurrency: string
}): Promise<number>
```

Get the current exchange rate for a given currency.

#### getSwapQuote

```tsx
getSwapQuote(params: {
  chainId: number
  userAddress: string
  sellCurrencyAddress: string
  buyCurrencyAddress: string
  sellAmount: string
}): Promise<SwapQuote>
```

Get a quote for swapping between two tokens.
