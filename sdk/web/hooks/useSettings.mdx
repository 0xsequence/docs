## Import

```tsx
import { useSettings } from '@0xsequence/hooks'
```

## Usage

```tsx
import { useSettings } from '@0xsequence/hooks'

function WalletView() {
  const {
    hideCollectibles,
    fiatCurrency,
    selectedNetworks,
    setHideCollectibles
  } = useSettings()

  return (
    <div>
      <Toggle
        checked={hideCollectibles}
        onChange={setHideCollectibles}
        label="Hide NFTs"
      />
      <Text>Currency: {fiatCurrency.symbol}</Text>
    </div>
  )
}
```

## Return Type: `Settings`

The hook returns an object containing current settings and setter functions:

```tsx
interface Settings {
  // Current Settings
  hideCollectibles: boolean
  hideUnlistedTokens: boolean
  fiatCurrency: FiatCurrency
  selectedNetworks: number[]
  
  // Setter Functions
  setFiatCurrency: (newFiatCurrency: FiatCurrency) => void
  setHideCollectibles: (newState: boolean) => void
  setHideUnlistedTokens: (newState: boolean) => void
  setSelectedNetworks: (newNetworks: number[]) => void
}

interface FiatCurrency {
  decimals: number
  name: {
    message: string
  }
  sign: string
  symbol: string
}
```

### Properties

### Current Settings

#### hideCollectibles
`boolean`

Whether to hide NFT collectibles in the wallet view.

#### hideUnlistedTokens
`boolean`

Whether to hide unverified tokens. Defaults to `true`.

#### fiatCurrency
`FiatCurrency`

Selected fiat currency for value display (e.g., USD, EUR).

#### selectedNetworks
`number[]`

Array of chain IDs for networks to display.

#### Setter Functions

#### setFiatCurrency
`(newFiatCurrency: FiatCurrency) => void`

Updates the preferred fiat currency for value display.

#### setHideCollectibles
`(newState: boolean) => void`

Toggles the visibility of NFT collectibles.

#### setHideUnlistedTokens
`(newState: boolean) => void`

Toggles the visibility of unverified tokens.

#### setSelectedNetworks
`(newNetworks: number[]) => void`

Updates the list of selected networks to display.

## Examples

### Managing Token Visibility

```tsx
function TokenVisibilitySettings() {
  const { 
    hideUnlistedTokens, 
    setHideUnlistedTokens,
    hideCollectibles,
    setHideCollectibles
  } = useSettings()

  return (
    <div>
      <h2>Token Visibility</h2>
      
      <Toggle
        checked={hideUnlistedTokens}
        onChange={setHideUnlistedTokens}
        label="Hide Unverified Tokens"
      />
      
      <Toggle
        checked={hideCollectibles}
        onChange={setHideCollectibles}
        label="Hide NFTs"
      />
    </div>
  )
}
```

### Currency Selection

```tsx
function CurrencySettings() {
  const { fiatCurrency, setFiatCurrency } = useSettings()

  const currencies: FiatCurrency[] = [
    { symbol: 'USD' },
    { symbol: 'EUR' },
    { symbol: 'GBP' }
  ]

  return (
    <div>
      <h2>Display Currency</h2>
      <select
        value={fiatCurrency.symbol}
        onChange={(e) => {
          const newCurrency = currencies.find(c => c.symbol === e.target.value)
          if (newCurrency) {
            setFiatCurrency(newCurrency)
          }
        }}
      >
        {currencies.map(currency => (
          <option key={currency.symbol} value={currency.symbol}>
            {currency.symbol}
          </option>
        ))}
      </select>
    </div>
  )
}
```

### Network Selection

```tsx
function NetworkSettings() {
  const { selectedNetworks, setSelectedNetworks } = useSettings()

      setSelectedNetworks(selectedNetworks.filter(id => id !== chainId))

      setSelectedNetworks([...selectedNetworks, chainId])

```

## Storage Details

Settings are stored in localStorage under the `LocalStorageKey.Settings` key. The stored object has the following structure:

```tsx
{
  hideCollectibles: boolean
  hideUnlistedTokens: boolean
  fiatCurrency: FiatCurrency
  selectedNetworks: number[]
}
```

If storage access fails or values are invalid, the hook falls back to these defaults:
- `hideUnlistedTokens`: `true`
- `hideCollectibles`: `false`
- `fiatCurrency`: System default currency
- `selectedNetworks`: All available chains
