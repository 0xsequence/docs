<Steps>
<Step title="Install the required libraries">
```bash
    npm install @0xsequence/connect @0xsequence/checkout 
    # or
    pnpm install @0xsequence/connect @0xsequence/checkout 
    # or
    yarn add @0xsequence/connect @0xsequence/checkout 
```
</Step>

<Step title="Wrap your App with the Providers">

<Note>
    Make sure you have configured your App with Connect [v5](/sdk/web/wallet-sdk/embedded/getting-started) or [v6](/sdk/web/wallet-sdk/ecosystem/getting-started) before continuing.
</Note>

```typescript [main.tsx]
    import React from "react";
    import ReactDOM from "react-dom/client";
    import "./index.css";

    import App from "./App";
    import { config } from "./config";
    import { SequenceConnect } from "@0xsequence/connect";
    import { SequenceCheckoutProvider } from "@0xsequence/checkout";

    function Dapp() {
    return (
        <SequenceConnect config={config}>
            <SequenceCheckoutProvider>
                <App />
            </SequenceCheckoutProvider>
        </SequenceConnect>
    );
    }

    ReactDOM.createRoot(document.getElementById("root")!).render(
    <React.StrictMode>
        <Dapp />
    </React.StrictMode>
    );
```
</Step>

<Step title="Start using it">
    Now you can start using hooks like [useCheckoutModal](/sdk/web/checkout-sdk/hooks/useCheckoutModal).
</Step>

</Steps>
