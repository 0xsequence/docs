---
title: useCountOfCollectables
description: The useCountOfCollectables hook returns the number of NFTs in a collection.
sidebarTitle: useCountOfCollectables
---

```ts
import { OrderSide } from '@0xsequence/marketplace-sdk';
import { useCountOfCollectables } from '@0xsequence/marketplace-sdk/react';

const countOfCollectables = useCountOfCollectables({
  chainId,
  collectionAddress,
  side: OrderSide.listing,
  filter: {
    searchText: text,
    includeEmpty,
    properties,
  },
});
```
